using System;
namespace Programa{
class Program{
static void Main(){
double glic;

Console.WriteLine("Quantidade de Glicose: ");
glic = double.Parse(Console.ReadLine());

if (glic <= 100){
	Console.WriteLine("Nivel de Glicose: Normal");
}
else if (glic > 100 && glic <= 140){
	Console.WriteLine("Nivel de Glicose: Elevado");
}
else if (glic > 140){
	Console.WriteLine("Nivel de Glicose: Diabetes");
}

}
}
}
