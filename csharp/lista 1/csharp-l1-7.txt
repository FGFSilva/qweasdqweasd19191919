using System;
namespace Programa{
class Program{
static void Main(){
int destino, tipo;
float valorpago;

Console.WriteLine("Qual o destino pretendido?\n1-Morros\n2-Barreirinhas\n3-Bacabeira\n4-Rosario\n");
destino = int.Parse(Console.ReadLine());

switch (destino)
{
	case 1:
		valorpago = 120;
		break;
	case 2:
		valorpago = 200;
		break;
	case 3:
		valorpago = 50;
		break;
	case 4:
		valorpago = 80;
		break;
}

Console.WriteLine("Qual o tipo de viagem?\n1-Ida e volta\n2-So ida\n");
tipo = int.Parse(Console.ReadLine());

if (tipo == 1)
{
	valorpago = 0.9 * 2 * valorpago;
}

Console.WriteLine("O valor pago pela viagem e de R$ "+valorpago+"");

}
}
}
