using System;
namespace Programa{
class Program{
static void Main(){
double valor;
int minutos;

Console.WriteLine("Digite a quantidade de minutos consumida:");
minutos = int.Parse(Console.ReadLine());

if (minutos <= 100)
{
	Console.WriteLine("O valor da conta e R$ 50.00");
}
else
{
	valor = (minutos - 100) * 2.00 + 50;
	Console.WriteLine("O valor da conta e R$ "+valor+"");
}

}
}
}
