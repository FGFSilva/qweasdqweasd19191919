using System;
namespace Programa{
class Program{
static void Main(){
double larg,comp,area,valorm2,preco;

Console.WriteLine("Digite o valor do comprimento: ");
comp = double.Parse(Console.ReadLine());

Console.WriteLine("Digite o valor da largura: ");
larg = double.Parse(Console.ReadLine());

area = comp * larg;

Console.WriteLine("Digite o valor do metro quadrado: ");
valorm2 = double.Parse(Console.ReadLine());

preco = area * valorm2;

Console.WriteLine("A area do terreno e: %.2lf\n", area);
Console.WriteLine("O preco do terreno e: %.2lf\n", preco);

}
}
}
