using System;
namespace Programa{
class Program{
static void Main(){
double nota1, nota2, media;

Console.WriteLine("Digite a primeira nota: ");
nota1 = double.Parse(Console.ReadLine());

Console.WriteLine("Digite a segunda nota: ");
nota2 = double.Parse(Console.ReadLine());

media = (nota1 + nota2) / 2;

if (media < 6.0)
{
	Console.WriteLine("O aluno esta REPROVADO com media "+media+"");
}
else
{
	Console.WriteLine("O aluno esta APROVADO com media "+media+"");
}

}
}
}
